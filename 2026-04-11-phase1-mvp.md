**Goal:** Build the "Daily Briefing" script — a single command that pulls NSE EOD data, computes market breadth, runs three setup scans, and prints a ranked watchlist.

**Architecture:** Python + SQLite. No server, no UI, no broker connection. One script (`scripts/daily_briefing.py`) orchestrates everything. Each layer is a separate module with clear interfaces.

**Tech Stack:** Python 3.11, pandas, sqlite3, nsepy/yfinance, pytest, ta-lib (or pandas_ta as fallback)

**Success criteria:** `python scripts/daily_briefing.py` runs without error and prints the Daily Briefing output format defined in `AGENT_RULES.md`.

# src/ingestion/store.py
import sqlite3
import logging
from typing import List, Dict
import pandas as pd
import config

logger = logging.getLogger(__name__)


def get_connection() -> sqlite3.Connection:
    """Return a sqlite3 connection to the configured DB path."""
    return sqlite3.connect(config.DB_PATH)


def init_db() -> None:
    """Create all tables if they do not already exist."""
    conn = get_connection()
    try:
        cursor = conn.cursor()
        cursor.executescript("""
            CREATE TABLE IF NOT EXISTS symbols (
                symbol      TEXT PRIMARY KEY,
                name        TEXT,
                sector      TEXT,
                index_name  TEXT,
                active      INTEGER DEFAULT 1
            );

            CREATE TABLE IF NOT EXISTS ohlcv (
                id          INTEGER PRIMARY KEY AUTOINCREMENT,
                symbol      TEXT NOT NULL,
                date        TEXT NOT NULL,
                open        REAL,
                high        REAL,
                low         REAL,
                close       REAL,
                volume      INTEGER,
                UNIQUE(symbol, date),
                FOREIGN KEY (symbol) REFERENCES symbols(symbol)
            );

            CREATE INDEX IF NOT EXISTS idx_ohlcv_symbol_date ON ohlcv(symbol, date);
            CREATE INDEX IF NOT EXISTS idx_ohlcv_date ON ohlcv(date);

            CREATE TABLE IF NOT EXISTS breadth (
                date                TEXT PRIMARY KEY,
                pct_above_ma20      REAL,
                pct_above_ma50      REAL,
                new_highs_52w       INTEGER,
                new_lows_52w        INTEGER,
                up_volume_ratio     REAL,
                advancing           INTEGER,
                declining           INTEGER,
                verdict             TEXT
            );

            CREATE TABLE IF NOT EXISTS watchlist (
                id              INTEGER PRIMARY KEY AUTOINCREMENT,
                date            TEXT NOT NULL,
                symbol          TEXT NOT NULL,
                setup_type      TEXT NOT NULL,
                score           REAL,
                pct_change      REAL,
                volume_ratio    REAL,
                close           REAL,
                notes           TEXT
            );
        """)
        conn.commit()
        logger.info("Database initialized at %s", config.DB_PATH)
    except sqlite3.OperationalError as e:
        logger.error("Failed to initialize database: %s", e)
        raise
    finally:
        conn.close()


def upsert_ohlcv(rows: List[Dict]) -> int:
    """Insert or replace OHLCV rows. Returns count of rows processed."""
    if not rows:
        return 0
    conn = get_connection()
    try:
        cursor = conn.cursor()
        cursor.executemany(
            """INSERT OR REPLACE INTO ohlcv
               (symbol, date, open, high, low, close, volume)
               VALUES (:symbol, :date, :open, :high, :low, :close, :volume)""",
            rows
        )
        conn.commit()
        logger.debug("Upserted %d OHLCV rows", len(rows))
        return len(rows)
    except sqlite3.OperationalError as e:
        logger.error("Failed to upsert OHLCV rows: %s", e)
        raise
    finally:
        conn.close()


def get_ohlcv(symbol: str, days: int = 252) -> pd.DataFrame:
    """Return the last N trading days of OHLCV data for a symbol as a DataFrame."""
    conn = get_connection()
    try:
        df = pd.read_sql(
            """SELECT date, open, high, low, close, volume
               FROM ohlcv
               WHERE symbol = ?
               ORDER BY date DESC
               LIMIT ?""",
            conn, params=(symbol, days)
        )
        df['date'] = pd.to_datetime(df['date'])
        return df.sort_values('date').reset_index(drop=True)
    except sqlite3.OperationalError as e:
        logger.error("Failed to fetch OHLCV for %s: %s", symbol, e)
        raise
    finally:
        conn.close()


def get_all_symbols_ohlcv(date: str, lookback_days: int = 252) -> pd.DataFrame:
    """Return OHLCV for all symbols for the last N days ending on date."""
    conn = get_connection()
    try:
        df = pd.read_sql(
            """SELECT symbol, date, open, high, low, close, volume
               FROM ohlcv
               WHERE date <= ?
               ORDER BY symbol, date""",
            conn, params=(date,)
        )
        df['date'] = pd.to_datetime(df['date'])
        return df
    except sqlite3.OperationalError as e:
        logger.error("Failed to fetch all symbols OHLCV for %s: %s", date, e)
        raise
    finally:
        conn.close()


def save_breadth(record: Dict) -> None:
    """Insert or replace a daily breadth record."""
    conn = get_connection()
    try:
        conn.execute(
            """INSERT OR REPLACE INTO breadth
               (date, pct_above_ma20, pct_above_ma50, new_highs_52w, new_lows_52w,
                up_volume_ratio, advancing, declining, verdict)
               VALUES (:date, :pct_above_ma20, :pct_above_ma50, :new_highs_52w,
                       :new_lows_52w, :up_volume_ratio, :advancing, :declining, :verdict)""",
            record
        )
        conn.commit()
        logger.info("Saved breadth for %s: %s", record['date'], record['verdict'])
    except sqlite3.OperationalError as e:
        logger.error("Failed to save breadth for %s: %s", record.get('date'), e)
        raise
    finally:
        conn.close()


def save_watchlist(entries: List[Dict]) -> None:
    """Insert watchlist candidates for the day."""
    if not entries:
        return
    conn = get_connection()
    try:
        conn.executemany(
            """INSERT INTO watchlist
               (date, symbol, setup_type, score, pct_change, volume_ratio, close, notes)
               VALUES (:date, :symbol, :setup_type, :score, :pct_change,
                       :volume_ratio, :close, :notes)""",
            entries
        )
        conn.commit()
        logger.info("Saved %d watchlist entries", len(entries))
    except sqlite3.OperationalError as e:
        logger.error("Failed to save watchlist: %s", e)
        raise
    finally:
        conn.close()
```

# src/ingestion/fetcher.py
import logging
import time
from datetime import date, datetime
from typing import List, Dict, Optional
import requests
import yfinance as yf
import pandas as pd
import config

logger = logging.getLogger(__name__)

_NSE_BHAVCOPY_URL = (
    "https://www1.nseindia.com/content/historical/EQUITIES/{year}/{month}/cm{dd}{MON}{year}bhav.csv.zip"
)
_HEADERS = {
    'User-Agent': 'Mozilla/5.0 (X11; Linux x86_64) AppleWebKit/537.36',
    'Referer': 'https://www.nseindia.com',
    'Accept-Language': 'en-US,en;q=0.9',
}


def parse_bhavcopy_rows(rows: List[Dict], date: str) -> List[Dict]:
    """
    Parse a list of Bhavcopy CSV row dicts into OHLCV dicts.
    Only EQ series rows are included. Non-EQ (BE, BL, INDEX) are skipped.
    """
    result = []
    for row in rows:
        if row.get('SERIES', '').strip() != 'EQ':
            continue
        try:
            result.append({
                'symbol': row['SYMBOL'].strip(),
                'date': date,
                'open': float(row['OPEN']),
                'high': float(row['HIGH']),
                'low': float(row['LOW']),
                'close': float(row['CLOSE']),
                'volume': int(float(row['TOTTRDQTY'])),
            })
        except (KeyError, ValueError) as e:
            logger.warning("Skipping malformed row for %s: %s", row.get('SYMBOL', '?'), e)
    return result


def fetch_via_yfinance(symbols: List[str], fetch_date: str) -> List[Dict]:
    """
    Fetch EOD data for a list of NSE symbols using yfinance as fallback.
    fetch_date: ISO date string e.g. '2026-04-11'
    """
    results = []
    failed = []
    for symbol in symbols:
        try:
            ticker = yf.Ticker(f"{symbol}.NS")
            # Fetch a 5-day window to ensure we get the target date
            df = ticker.history(period="5d")
            if df.empty:
                logger.warning("yfinance: no data for %s", symbol)
                failed.append(symbol)
                continue
            df.index = pd.to_datetime(df.index).date
            target = datetime.strptime(fetch_date, '%Y-%m-%d').date()
            if target not in df.index:
                logger.warning("yfinance: date %s not found for %s", fetch_date, symbol)
                failed.append(symbol)
                continue
            row = df.loc[target]
            results.append({
                'symbol': symbol,
                'date': fetch_date,
                'open': round(float(row['Open']), 2),
                'high': round(float(row['High']), 2),
                'low': round(float(row['Low']), 2),
                'close': round(float(row['Close']), 2),
                'volume': int(row['Volume']),
            })
            time.sleep(0.1)  # Be polite to the API
        except Exception as e:
            logger.error("yfinance fetch failed for %s: %s", symbol, e)
            failed.append(symbol)

    if failed:
        logger.warning("yfinance: %d symbols failed: %s", len(failed), failed[:10])
    return results


def fetch_eod_data(symbols: List[str], fetch_date: Optional[str] = None) -> List[Dict]:
    """
    Main entry point. Fetch EOD OHLCV for given symbols on fetch_date.
    Tries yfinance (reliable, free). Returns list of OHLCV dicts.
    fetch_date defaults to today if not specified.
    """
    if fetch_date is None:
        fetch_date = date.today().isoformat()

    logger.info("Fetching EOD data for %d symbols on %s", len(symbols), fetch_date)
    results = fetch_via_yfinance(symbols, fetch_date)

    failure_rate = 1 - (len(results) / len(symbols)) if symbols else 0
    if failure_rate > 0.10:
        logger.error(
            "High failure rate: %.0f%% of symbols failed (%d/%d)",
            failure_rate * 100, len(symbols) - len(results), len(symbols)
        )
    else:
        logger.info("Fetch complete: %d/%d symbols succeeded", len(results), len(symbols))

    return results
```

def _make_ohlcv(n_symbols: int, n_days: int, pct_above_ma: float) -> pd.DataFrame:

# src/monitor/breadth.py
import logging
from typing import Dict
import pandas as pd
import numpy as np

logger = logging.getLogger(__name__)


def compute_breadth(df: pd.DataFrame) -> Dict:
    """
    Compute daily market breadth metrics from a multi-symbol OHLCV DataFrame.

    df must have columns: symbol, date, open, high, low, close, volume
    Returns a dict of breadth metrics for the most recent day in df.
    """
    if df.empty:
        logger.error("Empty DataFrame passed to compute_breadth")
        return {}

    df = df.copy()
    df['date'] = pd.to_datetime(df['date'])
    latest_date = df['date'].max()

    results = []
    for symbol, group in df.groupby('symbol'):
        g = group.sort_values('date').reset_index(drop=True)
        if len(g) < 21:
            continue  # Need at least 21 days for MA20

        g['ma20'] = g['close'].rolling(20).mean()
        g['ma50'] = g['close'].rolling(50).mean() if len(g) >= 51 else float('nan')

        # 52-week high/low (252 trading days)
        g['high_52w'] = g['high'].rolling(min(252, len(g))).max()
        g['low_52w'] = g['low'].rolling(min(252, len(g))).min()

        today = g[g['date'] == latest_date]
        if today.empty:
            continue

        row = today.iloc[0]
        prev = g[g['date'] < latest_date].iloc[-1] if len(g[g['date'] < latest_date]) > 0 else None

        results.append({
            'symbol': symbol,
            'close': row['close'],
            'prev_close': prev['close'] if prev is not None else row['close'],
            'volume': row['volume'],
            'above_ma20': int(not pd.isna(row['ma20']) and row['close'] > row['ma20']),
            'above_ma50': int(not pd.isna(row['ma50']) and row['close'] > row['ma50']),
            'new_52w_high': int(not pd.isna(row['high_52w']) and row['high'] >= row['high_52w']),
            'new_52w_low': int(not pd.isna(row['low_52w']) and row['low'] <= row['low_52w']),
        })

    if not results:
        logger.warning("No results computed in compute_breadth — insufficient data")
        return {}

    res = pd.DataFrame(results)
    total = len(res)
    advancing = int((res['close'] > res['prev_close']).sum())
    declining = int((res['close'] < res['prev_close']).sum())

    up_vol = res.loc[res['close'] >= res['prev_close'], 'volume'].sum()
    total_vol = res['volume'].sum()
    up_vol_ratio = round(up_vol / total_vol, 4) if total_vol > 0 else 0.0

    return {
        'date': latest_date.strftime('%Y-%m-%d'),
        'pct_above_ma20': round(res['above_ma20'].sum() / total * 100, 2),
        'pct_above_ma50': round(res['above_ma50'].sum() / total * 100, 2),
        'new_highs_52w': int(res['new_52w_high'].sum()),
        'new_lows_52w': int(res['new_52w_low'].sum()),
        'up_volume_ratio': up_vol_ratio,
        'advancing': advancing,
        'declining': declining,
    }

import logging
from typing import Dict
import config

logger = logging.getLogger(__name__)


def compute_verdict(metrics: Dict) -> str:
    """
    Given breadth metrics dict, return 'OFFENSIVE', 'DEFENSIVE', or 'AVOID'.
    Thresholds are read from config.py and are intentionally tunable.
    """
    if not metrics:
        logger.warning("Empty metrics passed to compute_verdict — returning AVOID")
        return 'AVOID'

    pct_ma20 = metrics.get('pct_above_ma20', 0)
    up_vol = metrics.get('up_volume_ratio', 0)
    highs = metrics.get('new_highs_52w', 0)
    lows = metrics.get('new_lows_52w', 1)  # Avoid division by zero

    # OFFENSIVE: broad participation, up-volume conviction, highs >> lows
    if (pct_ma20 >= config.MM_OFFENSIVE_MA20_PCT
            and up_vol >= config.MM_OFFENSIVE_UPVOL_RATIO
            and highs >= lows * config.MM_OFFENSIVE_HIGHS_VS_LOWS):
        return 'OFFENSIVE'

    # DEFENSIVE: mixed signals — some participation but not convincing
    if pct_ma20 >= config.MM_DEFENSIVE_MA20_PCT and highs >= lows:
        return 'DEFENSIVE'

    # AVOID: weak breadth — protect capital
    return 'AVOID'

# src/scanner/momentum_burst.py
import logging
from typing import Optional
import pandas as pd
import config

logger = logging.getLogger(__name__)


def detect_momentum_burst(
    df: pd.DataFrame,
    min_pct: Optional[float] = None,
    max_pct: Optional[float] = None,
    min_vol_ratio: Optional[float] = None,
    lookback: Optional[int] = None,
    min_price: Optional[float] = None,
    min_avg_vol: Optional[int] = None,
    max_prior_run: Optional[float] = None,
) -> pd.DataFrame:
    """
    Scan a multi-symbol OHLCV DataFrame for Momentum Burst setups.

    A Momentum Burst is detected when:
    - Close has moved min_pct to max_pct in the last 'lookback' days
    - Today's volume is >= min_vol_ratio * 20-day average volume
    - Close price is >= min_price
    - 20-day average volume >= min_avg_vol
    - The stock has NOT already run more than max_prior_run% in the prior 10 days

    Returns a DataFrame of detected setups with: symbol, pct_change, volume_ratio, score, close, setup_type.
    """
    min_pct = min_pct or config.MB_MIN_PCT_CHANGE
    max_pct = max_pct or config.MB_MAX_PCT_CHANGE
    min_vol_ratio = min_vol_ratio or config.MB_MIN_VOLUME_RATIO
    lookback = lookback or config.MB_LOOKBACK_DAYS
    min_price = min_price or config.MB_MIN_PRICE
    min_avg_vol = min_avg_vol or config.MB_MIN_AVG_VOLUME
    max_prior_run = max_prior_run or config.MB_MAX_PRIOR_RUN

    df = df.copy()
    df['date'] = pd.to_datetime(df['date'])
    candidates = []

    required_days = max(20, lookback + 10 + 1)

    for symbol, group in df.groupby('symbol'):
        g = group.sort_values('date').reset_index(drop=True)
        if len(g) < required_days:
            continue

        today = g.iloc[-1]
        if today['close'] < min_price:
            continue

        # 20-day average volume (excluding today)
        avg_vol_20 = g.iloc[-21:-1]['volume'].mean()
        if avg_vol_20 < min_avg_vol:
            continue

        # Volume ratio
        vol_ratio = today['volume'] / avg_vol_20 if avg_vol_20 > 0 else 0
        if vol_ratio < min_vol_ratio:
            continue

        # Price change over lookback period
        ref_close = g.iloc[-(lookback + 1)]['close']
        pct_change = (today['close'] - ref_close) / ref_close * 100
        if not (min_pct <= pct_change <= max_pct):
            continue

        # Prior run check: what was the move in the 10 days before the lookback window?
        prior_start_idx = -(lookback + 11)
        prior_end_idx = -(lookback + 1)
        if len(g) > abs(prior_start_idx):
            prior_close_start = g.iloc[prior_start_idx]['close']
            prior_close_end = g.iloc[prior_end_idx]['close']
            prior_run = (prior_close_end - prior_close_start) / prior_close_start * 100
            if prior_run > max_prior_run:
                logger.debug("Skipping %s: already ran %.1f%% prior to burst", symbol, prior_run)
                continue

        score = round(pct_change * vol_ratio, 2)
        candidates.append({
            'symbol': symbol,
            'setup_type': 'MOMENTUM_BURST',
            'pct_change': round(pct_change, 2),
            'volume_ratio': round(vol_ratio, 2),
            'score': score,
            'close': round(today['close'], 2),
            'notes': f"{pct_change:.1f}% on {vol_ratio:.1f}x volume",
        })

    if not candidates:
        return pd.DataFrame(columns=['symbol', 'setup_type', 'pct_change',
                                      'volume_ratio', 'score', 'close', 'notes'])

    result = pd.DataFrame(candidates)
    return result.sort_values('score', ascending=False).reset_index(drop=True)

import logging
from typing import Optional
import pandas as pd
import config

logger = logging.getLogger(__name__)


def detect_episodic_pivot(
    df: pd.DataFrame,
    min_gap_pct: Optional[float] = None,
    min_gap_vol_ratio: Optional[float] = None,
    max_days_since_gap: Optional[int] = None,
) -> pd.DataFrame:
    """
    Scan for Episodic Pivot setups: stocks that gapped up significantly
    on high volume and are still holding above the gap.
    """
    min_gap_pct = min_gap_pct or config.EP_MIN_GAP_PCT
    min_gap_vol_ratio = min_gap_vol_ratio or config.EP_MIN_GAP_VOLUME_RATIO
    max_days_since_gap = max_days_since_gap or config.EP_MAX_DAYS_SINCE_GAP

    df = df.copy()
    df['date'] = pd.to_datetime(df['date'])
    candidates = []

    for symbol, group in df.groupby('symbol'):
        g = group.sort_values('date').reset_index(drop=True)
        if len(g) < 25:
            continue

        today = g.iloc[-1]
        avg_vol_20 = g.iloc[-21:-1]['volume'].mean()
        if avg_vol_20 == 0:
            continue

        # Look back max_days_since_gap days for a gap event
        lookback_rows = g.iloc[-(max_days_since_gap + 1):-1]
        for idx in range(len(lookback_rows) - 1, -1, -1):
            gap_row = lookback_rows.iloc[idx]
            if idx == 0:
                prev_close = g.iloc[-(max_days_since_gap + 2)]['close'] if len(g) > max_days_since_gap + 2 else None
            else:
                prev_close = lookback_rows.iloc[idx - 1]['close']

            if prev_close is None:
                continue

            gap_pct = (gap_row['open'] - prev_close) / prev_close * 100
            gap_vol_ratio = gap_row['volume'] / avg_vol_20

            if gap_pct >= min_gap_pct and gap_vol_ratio >= min_gap_vol_ratio:
                # Is the stock still holding above the gap open?
                if today['close'] >= gap_row['open']:
                    days_since = len(g) - 1 - (len(g) - max_days_since_gap - 1 + idx)
                    score = round(gap_pct * gap_vol_ratio, 2)
                    candidates.append({
                        'symbol': symbol,
                        'setup_type': 'EPISODIC_PIVOT',
                        'pct_change': round(gap_pct, 2),
                        'volume_ratio': round(gap_vol_ratio, 2),
                        'score': score,
                        'close': round(today['close'], 2),
                        'notes': f"Gap {gap_pct:.1f}% {days_since}d ago on {gap_vol_ratio:.1f}x vol",
                    })
                break  # Only report the most recent gap

    if not candidates:
        return pd.DataFrame(columns=['symbol', 'setup_type', 'pct_change',
                                      'volume_ratio', 'score', 'close', 'notes'])

    return pd.DataFrame(candidates).sort_values('score', ascending=False).reset_index(drop=True)

import logging
from typing import Optional
import pandas as pd
import config

logger = logging.getLogger(__name__)


def detect_trend_intensity(
    df: pd.DataFrame,
    high_lookback: Optional[int] = None,
    ma50_trend_lookback: Optional[int] = None,
    min_days_above_ma50: Optional[int] = None,
    min_vol_ratio: Optional[float] = None,
    max_atr_pct: Optional[float] = None,
) -> pd.DataFrame:
    """
    Scan for Trend Intensity Breakout setups: steady uptrends breaking to new highs.
    """
    high_lookback = high_lookback or config.TI_HIGH_LOOKBACK_DAYS
    ma50_trend_lookback = ma50_trend_lookback or config.TI_MA50_TREND_LOOKBACK
    min_days_above_ma50 = min_days_above_ma50 or config.TI_MIN_DAYS_ABOVE_MA50
    min_vol_ratio = min_vol_ratio or config.TI_MIN_VOLUME_RATIO
    max_atr_pct = max_atr_pct or config.TI_MAX_ATR_PCT

    df = df.copy()
    df['date'] = pd.to_datetime(df['date'])
    candidates = []

    required_days = max(52, high_lookback + 5)

    for symbol, group in df.groupby('symbol'):
        g = group.sort_values('date').reset_index(drop=True)
        if len(g) < required_days:
            continue

        g['ma50'] = g['close'].rolling(50).mean()
        g['atr'] = (g['high'] - g['low']).rolling(14).mean()

        today = g.iloc[-1]
        if pd.isna(today['ma50']):
            continue

        # MA50 must be rising
        ma50_now = today['ma50']
        ma50_then = g.iloc[-ma50_trend_lookback - 1]['ma50']
        if pd.isna(ma50_then) or ma50_now <= ma50_then:
            continue

        # Today must be a new N-day high
        lookback_highs = g.iloc[-(high_lookback + 1):-1]['high'].max()
        if today['close'] <= lookback_highs:
            continue

        # Sufficient days above MA50 in last 50 days
        last_50 = g.iloc[-51:-1]
        days_above = (last_50['close'] > last_50['ma50']).sum()
        if days_above < min_days_above_ma50:
            continue

        # Volume confirmation
        avg_vol_20 = g.iloc[-21:-1]['volume'].mean()
        if avg_vol_20 == 0:
            continue
        vol_ratio = today['volume'] / avg_vol_20
        if vol_ratio < min_vol_ratio:
            continue

        # Low volatility (quiet trend)
        atr_pct = today['atr'] / today['close'] if today['close'] > 0 else 1.0
        if not pd.isna(atr_pct) and atr_pct > max_atr_pct:
            continue

        score = round(vol_ratio * (ma50_now - ma50_then) / ma50_then * 100, 2)
        candidates.append({
            'symbol': symbol,
            'setup_type': 'TREND_INTENSITY',
            'pct_change': round((today['close'] - lookback_highs) / lookback_highs * 100, 2),
            'volume_ratio': round(vol_ratio, 2),
            'score': score,
            'close': round(today['close'], 2),
            'notes': f"New {high_lookback}d high, {days_above}d above MA50",
        })

    if not candidates:
        return pd.DataFrame(columns=['symbol', 'setup_type', 'pct_change',
                                      'volume_ratio', 'score', 'close', 'notes'])

    return pd.DataFrame(candidates).sort_values('score', ascending=False).reset_index(drop=True)

# src/scanner/watchlist.py
import logging
import os
from datetime import date
from typing import List
import pandas as pd
import config
from src.ingestion.store import save_watchlist

logger = logging.getLogger(__name__)


def merge_and_rank(results: List[pd.DataFrame], scan_date: str = None) -> pd.DataFrame:
    """
    Merge results from multiple scanners, deduplicate, rank by score.
    A symbol can only appear once (take its highest-scoring setup).
    """
    if scan_date is None:
        scan_date = date.today().isoformat()

    all_results = pd.concat([r for r in results if not r.empty], ignore_index=True)
    if all_results.empty:
        return all_results

    all_results['date'] = scan_date
    # Keep highest score per symbol
    all_results = (all_results
                   .sort_values('score', ascending=False)
                   .drop_duplicates(subset='symbol', keep='first')
                   .reset_index(drop=True))
    return all_results.head(config.MAX_WATCHLIST_SIZE)


def export_watchlist(df: pd.DataFrame, scan_date: str = None) -> str:
    """Save watchlist to CSV and SQLite. Returns the CSV file path."""
    if scan_date is None:
        scan_date = date.today().isoformat()

    os.makedirs(config.WATCHLIST_DIR, exist_ok=True)
    csv_path = os.path.join(config.WATCHLIST_DIR, f"{scan_date}.csv")
    df.to_csv(csv_path, index=False)
    logger.info("Watchlist saved to %s", csv_path)

    records = df.to_dict(orient='records')
    save_watchlist(records)
    return csv_path
```

#!/usr/bin/env python3
# scripts/daily_briefing.py
"""
Dhanustambha Daily Briefing
Run every weekday after 16:30 IST to get the market verdict and setup watchlist.
"""
import logging
import os
import sys
from datetime import date

sys.path.insert(0, os.path.dirname(os.path.dirname(os.path.abspath(__file__))))

import config
from src.ingestion.store import init_db, get_all_symbols_ohlcv
from src.ingestion.fetcher import fetch_eod_data
from src.ingestion.store import upsert_ohlcv
from src.ingestion.symbols import get_universe_symbols
from src.monitor.breadth import compute_breadth
from src.monitor.verdict import compute_verdict
from src.scanner.momentum_burst import detect_momentum_burst
from src.scanner.episodic_pivot import detect_episodic_pivot
from src.scanner.trend_intensity import detect_trend_intensity
from src.scanner.watchlist import merge_and_rank, export_watchlist

logging.basicConfig(
    level=logging.INFO,
    format='%(asctime)s  %(levelname)-8s  %(message)s',
    handlers=[
        logging.FileHandler(os.path.join(config.LOG_DIR, 'briefing.log')),
        logging.StreamHandler(sys.stdout),
    ]
)
logger = logging.getLogger(__name__)


def run_briefing(fetch_date: str = None):
    """Main orchestration function for the daily briefing."""
    if fetch_date is None:
        fetch_date = date.today().isoformat()

    print(f"\n{'='*55}")
    print(f"  Dhanustambha Daily Briefing — {fetch_date}")
    print(f"{'='*55}")

    # 1. Ensure DB is initialised
    os.makedirs(os.path.dirname(config.DB_PATH), exist_ok=True)
    os.makedirs(config.LOG_DIR, exist_ok=True)
    init_db()

    # 2. Fetch today's EOD data
    symbols = get_universe_symbols(config.UNIVERSE)
    print(f"\n[1/4] Fetching EOD data for {len(symbols)} symbols...")
    rows = fetch_eod_data(symbols, fetch_date)
    if rows:
        upsert_ohlcv(rows)
        print(f"      ✓ {len(rows)} symbols fetched and stored")
    else:
        print("      ✗ No data fetched — is today a market holiday?")
        return

    # 3. Market Monitor
    print("\n[2/4] Computing Market Monitor breadth...")
    all_data = get_all_symbols_ohlcv(fetch_date)
    metrics = compute_breadth(all_data)
    verdict = compute_verdict(metrics)

    verdict_icon = {'OFFENSIVE': '✅', 'DEFENSIVE': '⚠️ ', 'AVOID': '🔴'}.get(verdict, '?')
    print(f"\n  {'─'*45}")
    print(f"  MARKET MONITOR")
    print(f"  {'─'*45}")
    print(f"  Stocks above MA20:    {metrics.get('pct_above_ma20', 0):.1f}%")
    print(f"  Stocks above MA50:    {metrics.get('pct_above_ma50', 0):.1f}%")
    print(f"  New 52w highs:        {metrics.get('new_highs_52w', 0)}")
    print(f"  New 52w lows:         {metrics.get('new_lows_52w', 0)}")
    print(f"  Up-volume ratio:      {metrics.get('up_volume_ratio', 0):.2f}")
    print(f"  Advancing / Declining:{metrics.get('advancing', 0)} / {metrics.get('declining', 0)}")
    print(f"  Verdict:              {verdict} {verdict_icon}")

    if verdict == 'AVOID':
        print("\n  🔴 AVOID — No new trades. Protect capital.")
        print(f"{'='*55}\n")
        return

    # 4. Run scanners
    print(f"\n[3/4] Running setup scanners (market is {verdict})...")
    mb_results = detect_momentum_burst(all_data)
    ep_results = detect_episodic_pivot(all_data)
    ti_results = detect_trend_intensity(all_data)
    print(f"      Momentum Burst:    {len(mb_results)} candidates")
    print(f"      Episodic Pivot:    {len(ep_results)} candidates")
    print(f"      Trend Intensity:   {len(ti_results)} candidates")

    # 5. Merge, rank, export
    print("\n[4/4] Building watchlist...")
    watchlist = merge_and_rank([mb_results, ep_results, ti_results], fetch_date)
    if watchlist.empty:
        print("      No candidates today.")
        print(f"{'='*55}\n")
        return

    csv_path = export_watchlist(watchlist, fetch_date)

    # Print the watchlist
    print(f"\n  {'─'*45}")
    print(f"  TOP CANDIDATES ({len(watchlist)})")
    print(f"  {'─'*45}")
    print(f"  {'SYMBOL':<12} {'SETUP':<18} {'%CHG':>6} {'VOL_RATIO':>10} {'PRICE':>8}")
    print(f"  {'─'*55}")
    for _, row in watchlist.iterrows():
        print(f"  {row['symbol']:<12} {row['setup_type']:<18} "
              f"{row['pct_change']:>+6.1f}% {row['volume_ratio']:>9.1f}x "
              f"{row['close']:>8.2f}")
    print(f"\n  Watchlist saved → {csv_path}")
    print(f"{'='*55}\n")


if __name__ == '__main__':
    import argparse
    parser = argparse.ArgumentParser(description='Dhanustambha Daily Briefing')
    parser.add_argument('--date', default=None, help='Override date (YYYY-MM-DD)')
    args = parser.parse_args()
    run_briefing(fetch_date=args.date)
```
